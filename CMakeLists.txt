cmake_minimum_required(VERSION 3.28)

include(${CMAKE_SOURCE_DIR}/cmake/pr_common.cmake)

option(UTIL_UDM_STATIC "Build as static library?" ON)

if(${UTIL_UDM_STATIC})
	set(LIB_TYPE STATIC)
else()
	set(LIB_TYPE SHARED)
endif()

set(PROJ_NAME util_udm)
pr_add_library(${PROJ_NAME} ${LIB_TYPE})


add_subdirectory("../pragma/external_libs/sharedutils-static")
add_subdirectory("../pragma/external_libs/sharedutils")
add_subdirectory("../pragma/external_libs/mathutil")
add_subdirectory("../pragma/external_libs/vfilesystem")


pr_add_dependency(${PROJ_NAME} vfilesystem TARGET PUBLIC)
pr_add_dependency(${PROJ_NAME} lz4 TARGET)

pr_add_headers(${PROJ_NAME} "include/")
pr_add_sources(${PROJ_NAME} "src/")

if(NOT ${UTIL_UDM_STATIC})
	pr_add_compile_definitions(
		${PROJ_NAME}
			-DUDM_SHARED
			-DUDM_EXPORT
			-DSHUTIL_STATIC
			-DMUTIL_STATIC
			-DVFILESYSTEM_STATIC
	)
endif()

if(MSVC)
	target_compile_options(${PROJ_NAME} PRIVATE "/bigobj")
else()
	target_link_options(${PROJ_NAME} PRIVATE "LINKER:--strip-all")
endif()

pr_finalize(${PROJ_NAME})
